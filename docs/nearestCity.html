      def findLocation(pos):
         city = ''
         x = pos[0]
         y = pos[1]
         minDistance = 1000
         for key in locations:
            pos = locations[key]
            difference = abs ( x - pos[0]) + abs (y - pos[1])
            if difference < minDistance:                                 
               city = key
               minDistance = difference
         return city
